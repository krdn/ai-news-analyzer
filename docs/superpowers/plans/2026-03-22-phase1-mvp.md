# Phase 1: 기반 구축 + 네이버 뉴스 MVP 구현 계획

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 네이버 뉴스 하나의 소스로 전체 파이프라인(수집→분석→시각화)을 검증하는 MVP 구축

**Architecture:** Next.js 16 App Router + BullMQ 워커 + PostgreSQL + Redis. 프론트엔드와 API는 Next.js에서 처리하고, 크롤링/분석은 별도 워커 프로세스로 분리. Docker Compose로 홈서버(192.168.0.5)에 배포.

**Tech Stack:** Next.js 16, TypeScript, Prisma, BullMQ, Redis, PostgreSQL, shadcn/ui, Tailwind CSS, Recharts, SWR, Zod, Axios, Cheerio, Vitest

**Spec:** `docs/superpowers/specs/2026-03-22-celeb-news-analyzer-design.md`

```prisma
// prisma/schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

enum CelebrityCategory {
  POLITICIAN
  ENTERTAINER
  OTHER
}

enum SourceType {
  ALL
  NAVER
  YOUTUBE
  X
  META
  COMMUNITY
}

enum SentimentLabel {
  VERY_POSITIVE
  POSITIVE
  NEUTRAL
  NEGATIVE
  VERY_NEGATIVE
}

enum AnalysisDepth {
  BASIC
  DEEP
}

enum PeriodType {
  HOURLY
  DAILY
  WEEKLY
}

model Celebrity {
  id           String            @id @default(uuid())
  name         String            @db.VarChar(100)
  aliases      String[]
  category     CelebrityCategory
  profileImage String?           @map("profile_image")
  description  String?
  createdAt    DateTime          @default(now()) @map("created_at")
  updatedAt    DateTime          @updatedAt @map("updated_at")

  sources   CelebritySource[]
  articles  Article[]
  snapshots SentimentSnapshot[]

  @@map("celebrities")
}

model CelebritySource {
  id             String     @id @default(uuid())
  celebrityId    String     @map("celebrity_id")
  sourceType     SourceType @map("source_type")
  searchKeywords String[]   @map("search_keywords")
  enabled        Boolean    @default(true)

  celebrity Celebrity @relation(fields: [celebrityId], references: [id], onDelete: Cascade)

  @@map("celebrity_sources")
}

model Article {
  id             String         @id @default(uuid())
  celebrityId    String         @map("celebrity_id")
  sourceType     SourceType     @map("source_type")
  sourceUrl      String         @unique @map("source_url")
  title          String
  content        String?
  author         String?        @db.VarChar(200)
  publishedAt    DateTime?      @map("published_at")
  collectedAt    DateTime       @default(now()) @map("collected_at")
  sentimentScore Float?         @map("sentiment_score")
  sentimentLabel SentimentLabel? @map("sentiment_label")

  celebrity Celebrity @relation(fields: [celebrityId], references: [id], onDelete: Cascade)
  comments  Comment[]

  @@index([celebrityId, publishedAt])
  @@map("articles")
}

model Comment {
  id                  String         @id @default(uuid())
  articleId           String         @map("article_id")
  content             String
  author              String?        @db.VarChar(200)
  likes               Int            @default(0)
  publishedAt         DateTime?      @map("published_at")
  sentimentScore      Float?         @map("sentiment_score")
  sentimentConfidence Float?         @map("sentiment_confidence")
  sentimentLabel      SentimentLabel? @map("sentiment_label")
  emotions            String[]
  topics              String[]
  analysisDepth       AnalysisDepth? @map("analysis_depth")

  article Article @relation(fields: [articleId], references: [id], onDelete: Cascade)

  @@index([articleId])
  @@index([sentimentScore])
  @@index([publishedAt])
  @@map("comments")
}

model SentimentSnapshot {
  id            String      @id @default(uuid())
  celebrityId   String      @map("celebrity_id")
  periodType    PeriodType  @map("period_type")
  periodStart   DateTime    @map("period_start")
  sourceType    SourceType  @default(ALL) @map("source_type")
  totalComments Int         @default(0) @map("total_comments")
  avgScore      Float       @default(0) @map("avg_score")
  positiveCount Int         @default(0) @map("positive_count")
  neutralCount  Int         @default(0) @map("neutral_count")
  negativeCount Int         @default(0) @map("negative_count")
  topEmotions   Json?       @map("top_emotions")
  topTopics     Json?       @map("top_topics")

  celebrity Celebrity @relation(fields: [celebrityId], references: [id], onDelete: Cascade)

  @@unique([celebrityId, periodType, periodStart, sourceType], name: "snapshot_unique")
  @@index([celebrityId, periodType, periodStart])
  @@map("sentiment_snapshots")
}
```

```typescript
export const QUEUE_NAMES = {
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/shared/lib/prisma";
import { createCelebritySchema } from "@/entities/celebrity";

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  const query = searchParams.get("q");
  const category = searchParams.get("category");

  const where: Record<string, unknown> = {};

  if (query) {
    where.OR = [
      { name: { contains: query, mode: "insensitive" } },
      { aliases: { has: query } },
    ];
  }

  if (category) {
    where.category = category;
  }

  const celebrities = await prisma.celebrity.findMany({
    where,
    orderBy: { createdAt: "desc" },
  });

  return NextResponse.json(celebrities);
}

export async function POST(request: NextRequest) {
  const body = await request.json();
  const parsed = createCelebritySchema.safeParse(body);

  if (!parsed.success) {
    return NextResponse.json(
      { error: "유효하지 않은 데이터", details: parsed.error.flatten() },
      { status: 400 }
    );
  }

  const celebrity = await prisma.celebrity.create({
    data: parsed.data,
  });

  return NextResponse.json(celebrity, { status: 201 });
}
```

```typescript
// src/workers/crawler/naver.ts
import axios from "axios";
import * as cheerio from "cheerio";
import { prisma } from "@/shared/lib/prisma";
import { analysisQueue } from "@/shared/lib/queue";

interface NaverSearchItem {
  title: string;
  link: string;
  description: string;
  pubDate: string;
}

interface NaverSearchResponse {
  items: NaverSearchItem[];
}

interface ParsedArticle {
  celebrityId: string;
  sourceType: "NAVER";
  sourceUrl: string;
  title: string;
  content: string;
  publishedAt: Date;
}

function stripHtml(html: string): string {
  const $ = cheerio.load(`<span>${html}</span>`);
  return $("span").text();
}

export function parseNaverSearchResponse(
  response: NaverSearchResponse,
  celebrityId: string
): ParsedArticle[] {
  return response.items.map((item) => ({
    celebrityId,
    sourceType: "NAVER" as const,
    sourceUrl: item.link,
    title: stripHtml(item.title),
    content: stripHtml(item.description),
    publishedAt: new Date(item.pubDate),
  }));
}

export async function fetchNaverNews(
  query: string,
  celebrityId: string
): Promise<ParsedArticle[]> {
  const response = await axios.get<NaverSearchResponse>(
    "https://openapi.naver.com/v1/search/news.json",
    {
      params: { query, display: 20, sort: "date" },
      headers: {
        "X-Naver-Client-Id": process.env.NAVER_CLIENT_ID,
        "X-Naver-Client-Secret": process.env.NAVER_CLIENT_SECRET,
      },
    }
  );

  return parseNaverSearchResponse(response.data, celebrityId);
}

export interface ParsedComment {
  content: string;
  author: string;
  likes: number;
  publishedAt: Date;
}

export function parseNaverComments(rawComments: Array<{
  contents: string;
  userName: string;
  sympathyCount: number;
  modTime: string;
}>): ParsedComment[] {
  return rawComments.map((c) => ({
    content: c.contents,
    author: c.userName,
    likes: c.sympathyCount,
    publishedAt: new Date(c.modTime),
  }));
}

export async function crawlNaverForCelebrity(celebrityId: string) {
  const celebrity = await prisma.celebrity.findUnique({
    where: { id: celebrityId },
    include: {
      sources: { where: { sourceType: "NAVER", enabled: true } },
    },
  });

  if (!celebrity) return;

  const keywords =
    celebrity.sources[0]?.searchKeywords ?? [celebrity.name, ...celebrity.aliases];

  for (const keyword of keywords) {
    const articles = await fetchNaverNews(keyword, celebrityId);

    for (const article of articles) {
      const existing = await prisma.article.findUnique({
        where: { sourceUrl: article.sourceUrl },
      });

      if (existing) continue;

      const saved = await prisma.article.create({ data: article });

      // 댓글 수집
      const comments = await fetchNaverComments(article.sourceUrl);
      if (comments.length > 0) {
        await prisma.comment.createMany({
          data: comments.map((c) => ({
            articleId: saved.id,
            content: c.content,
            author: c.author,
            likes: c.likes,
            publishedAt: c.publishedAt,
          })),
        });
      }

      // 분석 큐에 추가
      await analysisQueue.add("analyze-article", {
        articleId: saved.id,
        celebrityId,
      });
    }
  }
}

// 네이버 뉴스 댓글 수집 (비공식 API)
export async function fetchNaverComments(articleUrl: string): Promise<ParsedComment[]> {
  try {
    // 네이버 뉴스 URL에서 oid/aid 추출
    const match = articleUrl.match(/\/article\/(\d+)\/(\d+)/);
    if (!match) return [];

    const [, oid, aid] = match;
    const response = await axios.get(
      `https://apis.naver.com/commentBox/cbox/web_naver_list_jsonp.json`,
      {
        params: {
          ticket: "news",
          templateId: "default_society",
          pool: "cbox5",
          lang: "ko",
          country: "KR",
          objectId: `news${oid},${aid}`,
          pageSize: 50,
          indexSize: 10,
          page: 1,
          sort: "FAVORITE",
        },
        headers: {
          Referer: articleUrl,
        },
      }
    );

    const comments = response.data?.result?.commentList ?? [];
    return parseNaverComments(comments);
  } catch {
    // 댓글 수집 실패 시 빈 배열 반환 (기사는 이미 저장됨)
    return [];
  }
}
```

```typescript
// src/workers/analyzer/sentiment.ts
import type { SentimentLabel } from "@prisma/client";

const POSITIVE_WORDS = [
  "좋다", "좋아", "최고", "대단", "멋지", "훌륭", "감동", "사랑",
  "응원", "축하", "대박", "짱", "존경", "감사", "행복", "기쁘",
  "잘했", "잘한", "굿", "브라보", "멋있", "예쁘", "귀엽",
];

const NEGATIVE_WORDS = [
  "나쁘", "싫다", "싫어", "최악", "실망", "짜증", "화나", "분노",
  "별로", "쓰레기", "한심", "답답", "혐오", "역겹", "못생",
  "거짓", "사기", "비호감", "끔찍", "불쾌", "논란", "비판",
];

export interface SentimentResult {
  score: number;
  label: SentimentLabel;
  confidence: number;
}

export function analyzeSentiment(text: string): SentimentResult {
  if (!text.trim()) {
    return { score: 0, label: "NEUTRAL", confidence: 0 };
  }

  const lowerText = text.toLowerCase();
  let positiveCount = 0;
  let negativeCount = 0;

  for (const word of POSITIVE_WORDS) {
    if (lowerText.includes(word)) positiveCount++;
  }

  for (const word of NEGATIVE_WORDS) {
    if (lowerText.includes(word)) negativeCount++;
  }

  const total = positiveCount + negativeCount;

  if (total === 0) {
    return { score: 0, label: "NEUTRAL", confidence: 0.3 };
  }

  const rawScore = (positiveCount - negativeCount) / total;
  const score = Math.max(-1, Math.min(1, rawScore));
  const confidence = Math.min(1, total / 5);

  let label: SentimentLabel;
  if (score >= 0.6) label = "VERY_POSITIVE";
  else if (score >= 0.2) label = "POSITIVE";
  else if (score > -0.2) label = "NEUTRAL";
  else if (score > -0.6) label = "NEGATIVE";
  else label = "VERY_NEGATIVE";

  return { score, label, confidence };
}
```

```typescript
// src/workers/snapshot/aggregator.ts
import type { SentimentLabel } from "@prisma/client";

interface CommentForAggregation {
  sentimentLabel: SentimentLabel | null;
  sentimentScore: number | null;
  emotions: string[];
  topics: string[];
}

interface AggregationResult {
  totalComments: number;
  avgScore: number;
  positiveCount: number;
  neutralCount: number;
  negativeCount: number;
  topEmotions: Record<string, number>;
  topTopics: Record<string, number>;
}

export function aggregateComments(comments: CommentForAggregation[]): AggregationResult {
  if (comments.length === 0) {
    return {
      totalComments: 0,
      avgScore: 0,
      positiveCount: 0,
      neutralCount: 0,
      negativeCount: 0,
      topEmotions: {},
      topTopics: {},
    };
  }

  let positiveCount = 0;
  let neutralCount = 0;
  let negativeCount = 0;
  let scoreSum = 0;
  const emotionCounts: Record<string, number> = {};
  const topicCounts: Record<string, number> = {};

  for (const comment of comments) {
    const score = comment.sentimentScore ?? 0;
    scoreSum += score;

    const label = comment.sentimentLabel;
    if (label === "VERY_POSITIVE" || label === "POSITIVE") positiveCount++;
    else if (label === "VERY_NEGATIVE" || label === "NEGATIVE") negativeCount++;
    else neutralCount++;

    for (const emotion of comment.emotions) {
      emotionCounts[emotion] = (emotionCounts[emotion] ?? 0) + 1;
    }
    for (const topic of comment.topics) {
      topicCounts[topic] = (topicCounts[topic] ?? 0) + 1;
    }
  }

  return {
    totalComments: comments.length,
    avgScore: scoreSum / comments.length,
    positiveCount,
    neutralCount,
    negativeCount,
    topEmotions: emotionCounts,
    topTopics: topicCounts,
  };
}
```

```typescript
import { Worker } from "bullmq";
import { redis } from "../shared/lib/redis";
import { QUEUE_NAMES, snapshotQueue } from "../shared/lib/queue";
import { prisma } from "../shared/lib/prisma";
import { crawlNaverForCelebrity } from "./crawler/naver";
import { analyzeSentiment } from "./analyzer/sentiment";
import { aggregateComments } from "./snapshot/aggregator";

// 크롤링 워커
const crawlWorker = new Worker(
  QUEUE_NAMES.CRAWL,
  async (job) => {
    const { celebrityId } = job.data;
    console.log(`[CRAWL] 크롤링 시작: ${celebrityId}`);
    await crawlNaverForCelebrity(celebrityId);
    console.log(`[CRAWL] 크롤링 완료: ${celebrityId}`);
  },
  {
    connection: redis,
    concurrency: 2,
    limiter: { max: 1, duration: 2000 },
  }
);

// 분석 워커
const analysisWorker = new Worker(
  QUEUE_NAMES.ANALYSIS,
  async (job) => {
    const { articleId, celebrityId } = job.data;
    console.log(`[ANALYSIS] 분석 시작: article=${articleId}`);

    const comments = await prisma.comment.findMany({
      where: { articleId, sentimentScore: null },
    });

    for (const comment of comments) {
      const result = analyzeSentiment(comment.content);
      await prisma.comment.update({
        where: { id: comment.id },
        data: {
          sentimentScore: result.score,
          sentimentConfidence: result.confidence,
          sentimentLabel: result.label,
          analysisDepth: "BASIC",
        },
      });
    }

    // 스냅샷 집계 큐에 추가
    await snapshotQueue.add("aggregate", { celebrityId });
    console.log(`[ANALYSIS] 분석 완료: ${comments.length}개 댓글`);
  },
  { connection: redis, concurrency: 3 }
);

// 스냅샷 집계 워커
const snapshotWorker = new Worker(
  QUEUE_NAMES.SNAPSHOT,
  async (job) => {
    const { celebrityId } = job.data;
    const now = new Date();
    const hourStart = new Date(now.getFullYear(), now.getMonth(), now.getDate(), now.getHours());

    const comments = await prisma.comment.findMany({
      where: {
        article: { celebrityId },
        publishedAt: { gte: hourStart },
        sentimentLabel: { not: null },
      },
      select: {
        sentimentLabel: true,
        sentimentScore: true,
        emotions: true,
        topics: true,
      },
    });

    const aggregation = aggregateComments(comments);

    await prisma.sentimentSnapshot.upsert({
      where: {
        snapshot_unique: {
          celebrityId,
          periodType: "HOURLY",
          periodStart: hourStart,
          sourceType: "ALL",
        },
      },
      create: {
        celebrityId,
        periodType: "HOURLY",
        periodStart: hourStart,
        ...aggregation,
      },
      update: aggregation,
    });

    console.log(`[SNAPSHOT] 집계 완료: ${celebrityId}, ${aggregation.totalComments}개`);
  },
  { connection: redis, concurrency: 1 }
);

// 에러 핸들러
for (const worker of [crawlWorker, analysisWorker, snapshotWorker]) {
  worker.on("failed", (job, err) => {
    console.error(`[${worker.name}] 잡 실패: ${job?.id}`, err.message);
  });
}

console.log("워커 프로세스 시작됨");

// 그레이스풀 셧다운
process.on("SIGTERM", async () => {
  console.log("워커 종료 중...");
  await crawlWorker.close();
  await analysisWorker.close();
  await snapshotWorker.close();
  process.exit(0);
});
```

```typescript
    where: {
      celebrityId,
      periodType: period as "HOURLY" | "DAILY" | "WEEKLY",
      periodStart: { gte: since },
    },
    orderBy: { periodStart: "asc" },
  });

  const recentComments = await prisma.comment.findMany({
    where: {
      article: { celebrityId },
      sentimentLabel: { not: null },
    },
    orderBy: { publishedAt: "desc" },
    take: 20,
    include: {
      article: { select: { title: true, sourceUrl: true, sourceType: true } },
    },
  });

  return NextResponse.json({ snapshots, recentComments });
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/shared/lib/prisma";
import { crawlQueue } from "@/shared/lib/queue";

export async function POST(request: NextRequest) {
  const { celebrityId } = await request.json();

  if (celebrityId) {
    await crawlQueue.add("crawl-naver", { celebrityId });
    return NextResponse.json({ message: "크롤링 잡이 큐에 추가되었습니다" });
  }

  // 전체 셀럽 크롤링
  const celebrities = await prisma.celebrity.findMany({
    select: { id: true },
  });

  for (const celeb of celebrities) {
    await crawlQueue.add("crawl-naver", { celebrityId: celeb.id });
  }

  return NextResponse.json({
    message: `${celebrities.length}개 크롤링 잡이 큐에 추가되었습니다`,
  });
}
```
